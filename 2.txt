#include <iostream>
using namespace std;

char board[9] = {'1','2','3','4','5','6','7','8','9'};
char currentPlayer = 'X';

void displayBoard() {
    cout << " " << board[0] << " | " << board[1] << " | " << board[2] << endl;
    cout << "---|---|---" << endl;
    cout << " " << board[3] << " | " << board[4] << " | " << board[5] << endl;
    cout << "---|---|---" << endl;
    cout << " " << board[6] << " | " << board[7] << " | " << board[8] << endl;
}

bool checkWin() {
    for (int i = 0; i < 9; i += 3) // Check rows
        if (board[i] == board[i+1] && board[i+1] == board[i+2]) return true;
    for (int i = 0; i < 3; ++i) // Check columns
        if (board[i] == board[i+3] && board[i+3] == board[i+6]) return true;
    // Check diagonals
    return (board[0] == board[4] && board[4] == board[8]) || (board[2] == board[4] && board[4] == board[6]);
}

bool checkDraw() {
    for (int i = 0; i < 9; ++i) 
        if (board[i] != 'X' && board[i] != 'O') return false;
    return true;
}

void playTurn() {
    int move;
    cout << "Player " << currentPlayer << ", enter a position: ";
    cin >> move;
    if (board[move – 1] != 'X' && board[move – 1] != 'O') {
        board[move – 1] = currentPlayer;
        currentPlayer = (currentPlayer == 'X') ? 'O' : 'X';
    } else {
        cout << "Invalid move, try again.” << endl;
    }
}

int main() {
    while (true) {
        displayBoard();
        playTurn();
        if (checkWin()) {
            displayBoard();
            cout << "Player " << (currentPlayer == 'X' ? 'O' : 'X') << " wins!” << endl;
            break;
        }
        if (checkDraw()) {
            displayBoard();
            cout << "It's a draw!” << endl;
            break;
        }
    }
    return 0;
}
